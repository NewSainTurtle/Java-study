    }

    public List<Integer> bfs(Node root) {
        list = new ArrayList<>();
        Queue<Node> que = new LinkedList<>();
        que.offer(root);

        while (!que.isEmpty()) {
            Node node = que.poll();
            list.add(node.getValue());

            if (node.getLeft() != null) {
                que.offer(node.getLeft());
            }
            if (node.getRight() != null) {
                que.offer(node.getRight());
            }
        }
        
        return list;
    }
